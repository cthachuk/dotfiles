* Installation

#+begin_src sh

# simplify process by switching to root user
sudo bash
cd /var/tmp
nix-shell -p git
git clone https://github.com/cthachuk/dotfiles /mnt/etc/nixos
bash /var/tmp/dotfiles/hosts/enterprise/bootstrap.sh
reboot

#+end_src
